"""

# Crie uma função para somar dois números: 5+3

def somar():
    print(5+3)

somar()

def subtrair():

    print(5-3)

#Chamado da Função
subtrair()

def multiplicar():
    print(5*3)

multiplicar()

def dividir():
    print(5/3)

dividir()

    print(5//3)

dividir_inteiro()

def resto():
    print(5%3)

resto()

def exponenciação():
    print(5**3)

exponenciação()
